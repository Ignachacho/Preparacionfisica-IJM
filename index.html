// 🔊 SONIDO PRO

function beep(d=150,f=2000){

  const ctx=new (window.AudioContext||window.webkitAudioContext)();

  const osc=ctx.createOscillator();

  const gain=ctx.createGain();

  osc.type="square";

  osc.frequency.value=f;

  gain.gain.value=1.5;

  osc.connect(gain);

  gain.connect(ctx.destination);

  osc.start();

  setTimeout(()=>{osc.stop();ctx.close();},d);

}

function dobleBeep(){

  beep(150,1800);

  setTimeout(()=>beep(150,2300),120);

}

// FULLSCREEN AUTO

function fullscreen(){

  let el=document.documentElement;

  if(el.requestFullscreen) el.requestFullscreen();

}

// NAV

function ir(id){

  document.querySelectorAll(".pantalla").forEach(p=>p.classList.remove("activa"));

  document.getElementById(id).classList.add("activa");

  if(id==="biblioteca") cargarBiblioteca();

}

// ================= TIMER =================

let intervalo, tiempo, fase, ronda, pausado=false;

function parseTime(t){

  let [m,s]=t.split(":");

  return parseInt(m)*60+parseInt(s);

}

function format(t){

  let m=Math.floor(t/60);

  let s=t%60;

  return `${m.toString().padStart(2,"0")}:${s.toString().padStart(2,"0")}`;

}

function startTimer(){

  fullscreen();

  countdown(()=>{

    tiempo=parseTime(work.value);

    fase="work";

    ronda=1;

    pausado=false;

    ir("timer");

    intervalo=setInterval(()=>{

      if(pausado) return;

      tiempo--;

      if(tiempo<=3 && tiempo>0) beep();

      if(tiempo<=0){

        dobleBeep();

        if(fase==="work"){

          fase="rest";

          tiempo=parseTime(rest.value);

        } else {

          ronda++;

          if(ronda>rounds.value){

            clearInterval(intervalo);

            faseEl.textContent="FINAL";

            tiempoEl.textContent="00:00";

            return;

          }

          fase="work";

          tiempo=parseTime(work.value);

        }

      }

      update();

    },1000);

  });

}

function countdown(cb){

  let c=3;

  ir("timer");

  let cd=setInterval(()=>{

    tiempoEl.textContent=c;

    beep();

    c--;

    if(c<0){

      clearInterval(cd);

      dobleBeep();

      cb();

    }

  },1000);

}

function update(){

  tiempoEl.textContent=format(tiempo);

  faseEl.textContent=(fase==="work")?"TRABAJO":"DESCANSO";

  document.body.style.background=(fase==="work")?"#1565c0":"#ff6f00";

}

function parar(){ pausado=true; btnCont.style.display="inline-block"; }

function continuar(){ pausado=false; btnCont.style.display="none"; }

function salir(){ clearInterval(intervalo); ir("timerConfig"); document.body.style.background="#121212"; }

// guardar timer

function guardarTimer(){

  let arr=JSON.parse(localStorage.getItem("timers"))||[];

  arr.push({

    nombre:nombreTimer.value,

    work:work.value,

    rest:rest.value,

    rounds:rounds.value,

    fav:false

  });

  localStorage.setItem("timers",JSON.stringify(arr));

  alert("Guardado");

}

// ================= RANDOM =================

let intR, last=null, tiempoR=null, totalR=null, pausadoR=false;

function startRandom(){

  fullscreen();

  let dur=duracion.value ? parseTime(duracion.value) : null;

  totalR=dur;

  tiempoR=dur;

  ir("random");

  generar();

  intR=setInterval(()=>{

    if(pausadoR) return;

    generar();

    if(tiempoR!==null){

      tiempoR--;

      tiempoRandom.textContent=format(tiempoR);

      if(tiempoR<=0){

        clearInterval(intR);

        dobleBeep();

      }

    }

  },interval.value*1000);

}

function generar(){

  numero.textContent="";

  setTimeout(()=>{

    let n;

    do{

      n=Math.floor(Math.random()*(max.value-min.value+1))+parseInt(min.value);

    }while(noRepeat.checked && n===last);

    last=n;

    numero.textContent=n;

  },60);

}

function pararRandom(){ pausadoR=true; btnContR.style.display="inline-block"; }

function continuarRandom(){ pausadoR=false; btnContR.style.display="none"; }

function salirRandom(){ clearInterval(intR); ir("randomConfig"); }

function cargarBiblioteca(){

  let lista=document.getElementById("lista");

  lista.innerHTML="";

  let timers=JSON.parse(localStorage.getItem("timers"))||[];

  let randoms=JSON.parse(localStorage.getItem("randoms"))||[];

  timers.forEach((t,i)=>{

    lista.innerHTML+=`

    <div class="card">

      ⭐ ${t.fav?"✔️":""} <b>${t.nombre}</b><br>

      ${t.work}/${t.rest}/${t.rounds}

      <br>

      <button onclick="cargarTimer(${i})">CARGAR</button>

      <button onclick="favTimer(${i})">⭐</button>

      <button onclick="delTimer(${i})">🗑️</button>

    </div>`;

  });

  randoms.forEach((r,i)=>{

    lista.innerHTML+=`

    <div class="card">

      ⭐ ${r.fav?"✔️":""} <b>${r.nombre}</b><br>

      ${r.min}-${r.max} | ${r.interval}s

      <br>

      <button onclick="cargarRandom(${i})">CARGAR</button>

      <button onclick="favRandom(${i})">⭐</button>

      <button onclick="delRandom(${i})">🗑️</button>

    </div>`;

  });

}

// acciones biblioteca

function cargarTimer(i){

  let t=JSON.parse(localStorage.getItem("timers"))[i];

  nombreTimer.value=t.nombre;

  work.value=t.work;

  rest.value=t.rest;

  rounds.value=t.rounds;

  ir("timerConfig");

}

function favTimer(i){

  let arr=JSON.parse(localStorage.getItem("timers"));

  arr[i].fav=!arr[i].fav;

  localStorage.setItem("timers",JSON.stringify(arr));

  cargarBiblioteca();

}

function delTimer(i){

  let arr=JSON.parse(localStorage.getItem("timers"));

  arr.splice(i,1);

  localStorage.setItem("timers",JSON.stringify(arr));

  cargarBiblioteca();

}
